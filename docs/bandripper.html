342    def get_album_urls(self) -> list[str]:
343        """Get album urls from the main bandcamp url."""
344        if not self.discography_page_html:
345            self.discography_page_html = self.get_discography_page()
346        soup = BeautifulSoup(self.discography_page_html, "html.parser")
347        grid = soup.find("ol", attrs={"id": "music-grid"})
348        assert isinstance(grid, Tag)
349        parsed_url = urlparse(self.band_url)
350        base_url = f"https://{parsed_url.netloc}"
351        urls: list[str] = [
352            base_url + album.a.get("href") for album in grid.find_all("li")
353        ]
354        # Sometimes label pages link to a band's bandcamp instead of hosting the album
355        # so we gotta fix double urls
356        return [
357            url if url.count("http") == 1 else url[url.find("http", 1) :]
358            for url in urls
359        ]

361    def get_discography_page(self) -> str:
362        url = f"{colors.url}{self.band_url}[/]"
363        console.print(f"Fetching discography from {url}...")
364        response = None
365        try:
366            response = request(self.band_url)
367            response.raise_for_status()
368        except Exception as e:
369            console.print(f"Failed to access {url}.")
370            if response:
371                console.print(f"Status code: {colors.status}{response.status_code}")
372            raise e
373        return response.text

375    def save_discog_url(self) -> None:
376        """Add discog url to txt file if it isn't in there."""
377        discog_urls.touch()
378        urls = discog_urls.split("utf-8")
379        if self.band_url not in urls:
380            urls.append(self.band_url)
381        discog_urls.join(sorted(urls))

383    def rip(self) -> list[Album]:
384        """Rip all publicly available albums from the discography page.
385        Returns a list of `Album` objects that were ripped."""
386        console.print(f"Searching {colors.url}{self.band_url}[/] for albums...")
387        for url in self.get_album_urls():
388            self.album_rippers.append(
389                AlbumRipper(url, self.no_track_number, self.overwrite)
390            )
391        # Save discography url to list after successfully getting albums from it
392        self.save_discog_url()
393
394        console.print(f"Found {colors.number}{len(self.album_rippers)}[/] albums.")
395        console.print(f"Beginning rip...")
396        timer = Timer(subsecond_resolution=True)
397        timer.start()
398        fails: list[tuple[AlbumRipper, Exception]] = []
399        albums: list[Album] = []
400        for ripper in self.album_rippers:
401            try:
402                album = ripper.rip()
403                albums.append(album)
404            except Exception as e:
405                fails.append((ripper, e))
406        timer.stop()
407        console.print(
408            f"Finished downloading {colors.number}{len(self.album_rippers)}[/] albums in {colors.highlight}{timer.elapsed_str}."
409        )
410        if fails:
411            console.print(f"The following downloads failed:")
412            for fail in fails:
413                console.print(
414                    f"{colors.url}{fail[0].album_url}[/]: {colors.status}{fail[1]}"
415                )
416        return albums

220    def filter_existing_tracks(
221        self, tracks: list[Track]
222    ) -> tuple[list[Track], list[Track]]:
223        """Splits `tracks` into a list of tracks that don't exist at the save location and a list of tracks that do.

225        Return order is `(non_existing_tracks, existing_tracks)`."""

260    def rip(self) -> Album:
261        """Download and save the album tracks and album art."""
262        album = AlbumParser(self.get_album_page()).parse()
263        if not album:
264            raise RuntimeError(f"No album data was found on {self.album_url}.")
265        num_tracks = len(album.tracks)
266        if num_tracks == 0:
267            console.print(f"No public tracks available for {album.rich_str}.")
268            return album
269        self.make_save_path(album)
270        assert self.save_path
271        self.save_album_art(album)
272
273        tracks_to_download = self.get_track_download_list(album)
274        if not tracks_to_download:
275            return album
276
277        num_tracks_to_download = len(tracks_to_download)
278        quickpool.ThreadPool(
279            [self.save_track] * num_tracks_to_download,
280            [(track,) for track in tracks_to_download],
281            max_workers=5,
282        ).execute(
283            description=f"Downloading {colors.number}{num_tracks_to_download}[/] tracks from {album.rich_str}..."
284        )
285        console.print("Download complete.")
286        if self.failed_rips:
287            console.print("The following tracks failed to download:")
288            for track in self.failed_rips:
289                console.print(f"  {colors.title}{track.title}")
290        return album

292    def save_album_art(self, album: Album):
293        """Download and save album art if it has any."""
294        if not self.save_path:
295            raise self.NoSavePathError(album)
296        if album.art_url:
297            art = album.download_art()
298            if art:
299                (self.save_path / album.art_file_name).write_bytes(art)
300        else:
301            console.print(f"No album art detected for {album.rich_str}.")
302
303    def save_track(self, track: Track) -> bool:
304        """Save `track`.
305        Returns whether the download was successful or not."""
306        if not self.save_path:
307            raise self.NoSavePathError()
308        file_path = self.save_path / (
309            track.file_name if self.no_track_number else track.numbered_file_name
310        )
311        content = track.download()
312        if content:
313            file_path.write_bytes(content)
314            return True
315        self._failed_rips.append(track)
316        return False

318    def track_exists(self, track: Track) -> bool:
319        """Return if a track already exists in `self.save_path`."""

325        return path.exists()
